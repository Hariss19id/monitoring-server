## 🔌 VERCEL API ROUTES

### `GET /api/agent/[server_id]/[api_key]`
- Validate `server_id` + `api_key` dari Supabase
- Return: bash script dengan credentials ter-embed
- Content-Type: `text/plain`

### `POST /api/servers/register`
- Body: `{ name, os_type }`
- Generate: `server_id` (uuid) + `api_key` (random 32 char)
- Simpan ke Supabase tabel `servers`
- Return: `{ server_id, api_key, install_command }`

### `DELETE /api/servers/[server_id]`
- Hapus server dari Supabase
- Return: `{ success: true }`

## 📜 BASH AGENT SCRIPT TEMPLATE

File: `frontend/public/agent-template.sh`
(Di-serve oleh Vercel API Route dengan credentials di-inject)

```bash
#!/bin/bash
# ================================================
# Monitor Agent - Generated by monitoring-server
# Server ID : {{SERVER_ID}}
# Supabase  : {{SUPABASE_URL}}
# ================================================

SUPABASE_URL="{{SUPABASE_URL}}"
SUPABASE_KEY="{{SUPABASE_ANON_KEY}}"
SERVER_ID="{{SERVER_ID}}"
INTERVAL=5        # detik antar push
LOOP_COUNT=11     # 11 x 5 detik = 55 detik (1 menit cron)

# ─── Helper: POST ke Supabase REST ───────────────
supabase_insert() {
  local table=$1
  local payload=$2
  curl -s -X POST \
    "${SUPABASE_URL}/rest/v1/${table}" \
    -H "apikey: ${SUPABASE_KEY}" \
    -H "Authorization: Bearer ${SUPABASE_KEY}" \
    -H "Content-Type: application/json" \
    -H "Prefer: return=minimal" \
    -d "$payload" > /dev/null 2>&1
}

supabase_upsert() {
  local table=$1
  local payload=$2
  local on_conflict=$3
  curl -s -X POST \
    "${SUPABASE_URL}/rest/v1/${table}?on_conflict=${on_conflict}" \
    -H "apikey: ${SUPABASE_KEY}" \
    -H "Authorization: Bearer ${SUPABASE_KEY}" \
    -H "Content-Type: application/json" \
    -H "Prefer: resolution=merge-duplicates,return=minimal" \
    -d "$payload" > /dev/null 2>&1
}

# ─── Collector: CPU ──────────────────────────────
get_cpu_usage() {
  # Baca /proc/stat dua kali, hitung delta
  local cpu1=$(grep '^cpu ' /proc/stat)
  sleep 0.5
  local cpu2=$(grep '^cpu ' /proc/stat)

  local idle1=$(echo $cpu1 | awk '{print $5}')
  local total1=$(echo $cpu1 | awk '{s=0; for(i=2;i<=NF;i++) s+=$i; print s}')
  local idle2=$(echo $cpu2 | awk '{print $5}')
  local total2=$(echo $cpu2 | awk '{s=0; for(i=2;i<=NF;i++) s+=$i; print s}')

  local diff_idle=$((idle2 - idle1))
  local diff_total=$((total2 - total1))
  local usage=$(awk "BEGIN {printf \"%.1f\", (1 - $diff_idle/$diff_total) * 100}")
  echo "$usage"
}

get_cpu_temp() {
  local temp=0
  if [ -f /sys/class/thermal/thermal_zone0/temp ]; then
    temp=$(cat /sys/class/thermal/thermal_zone0/temp)
    temp=$(awk "BEGIN {printf \"%.1f\", $temp/1000}")
  fi
  echo "$temp"
}

get_cpu_model() {
  grep 'model name' /proc/cpuinfo | head -1 | cut -d: -f2 | xargs
}

get_cpu_cores() {
  nproc --all
}

# ─── Collector: RAM ──────────────────────────────
get_ram() {
  local mem_total=$(grep MemTotal /proc/meminfo | awk '{print $2}')
  local mem_free=$(grep MemAvailable /proc/meminfo | awk '{print $2}')
  local mem_used=$((mem_total - mem_free))
  local swap_total=$(grep SwapTotal /proc/meminfo | awk '{print $2}')
  local swap_free=$(grep SwapFree /proc/meminfo | awk '{print $2}')
  local swap_used=$((swap_total - swap_free))
  # Konversi ke bytes
  echo "$((mem_total*1024)) $((mem_used*1024)) $((mem_free*1024)) $((swap_total*1024)) $((swap_used*1024))"
}

# ─── Collector: Disk ─────────────────────────────
get_disk_json() {
  # Output JSON array dari df
  python3 -c "
import subprocess, json
result = subprocess.run(['df', '-B1', '--output=target,size,used,avail,pcent'], 
                       capture_output=True, text=True)
lines = result.stdout.strip().split('\n')[1:]
disks = []
for line in lines:
  parts = line.split()
  if len(parts) >= 5 and not parts[0].startswith('/dev/loop'):
    disks.append({
      'path': parts[0],
      'total': int(parts[1]),
      'used': int(parts[2]),
      'free': int(parts[3]),
      'use_pct': int(parts[4].replace('%',''))
    })
print(json.dumps(disks))
" 2>/dev/null || echo "[]"
}

# ─── Collector: Network ──────────────────────────
PREV_RX=0
PREV_TX=0

get_net_stats() {
  # Default interface (bukan lo)
  local iface=$(ip route | grep default | awk '{print $5}' | head -1)
  if [ -z "$iface" ]; then iface="eth0"; fi

  local rx=$(cat /proc/net/dev | grep " $iface:" | awk '{print $2}')
  local tx=$(cat /proc/net/dev | grep " $iface:" | awk '{print $10}')
  echo "$rx $tx $iface"
}

# ─── Collector: OS Info ──────────────────────────
get_os_info() {
  local hostname=$(hostname)
  local kernel=$(uname -r)
  local arch=$(uname -m)
  local distro="Unknown"
  local uptime_sec=$(awk '{print int($1)}' /proc/uptime)

  if [ -f /etc/os-release ]; then
    distro=$(grep PRETTY_NAME /etc/os-release | cut -d'"' -f2)
  fi

  echo "$hostname|$kernel|$arch|$distro|$uptime_sec"
}

# ─── Collector: Services ─────────────────────────
get_services_json() {
  if command -v systemctl &> /dev/null; then
    python3 -c "
import subprocess, json
result = subprocess.run(
  ['systemctl', 'list-units', '--type=service', '--no-pager', '--no-legend', '--all'],
  capture_output=True, text=True
)
services = []
for line in result.stdout.strip().split('\n')[:20]:
  parts = line.split()
  if len(parts) >= 4:
    services.append({
      'name': parts[0].replace('.service',''),
      'status': parts[3],
      'sub': parts[4] if len(parts) > 4 else ''
    })
print(json.dumps(services))
" 2>/dev/null || echo "[]"
  else
    echo "[]"
  fi
}

# ─── MAIN LOOP ───────────────────────────────────
# Jalankan setiap INTERVAL detik, total LOOP_COUNT kali
# Sehingga 1 eksekusi cron (1 menit) = 11 x push

# Update heartbeat & OS info (sekali per eksekusi)
OS_INFO=$(get_os_info)
OS_HOSTNAME=$(echo $OS_INFO | cut -d'|' -f1)
OS_KERNEL=$(echo $OS_INFO  | cut -d'|' -f2)
OS_ARCH=$(echo $OS_INFO    | cut -d'|' -f3)
OS_DISTRO=$(echo $OS_INFO  | cut -d'|' -f4)
OS_UPTIME=$(echo $OS_INFO  | cut -d'|' -f5)

# Update servers.last_seen dan os info
supabase_upsert "servers" "{
  \"id\": \"$SERVER_ID\",
  \"last_seen\": \"$(date -u +%Y-%m-%dT%H:%M:%SZ)\",
  \"hostname\": \"$OS_HOSTNAME\",
  \"kernel\": \"$OS_KERNEL\",
  \"distro\": \"$OS_DISTRO\",
  \"arch\": \"$OS_ARCH\",
  \"uptime_sec\": $OS_UPTIME
}" "id"

# Ambil network stats awal
NET_INIT=$(get_net_stats)
PREV_RX=$(echo $NET_INIT | awk '{print $1}')
PREV_TX=$(echo $NET_INIT | awk '{print $2}')
NET_IFACE=$(echo $NET_INIT | awk '{print $3}')

for i in $(seq 1 $LOOP_COUNT); do
  # ── CPU ──
  CPU_USAGE=$(get_cpu_usage)
  CPU_TEMP=$(get_cpu_temp)
  CPU_CORES=$(get_cpu_cores)

  # ── RAM ──
  RAM_DATA=$(get_ram)
  RAM_TOTAL=$(echo $RAM_DATA | awk '{print $1}')
  RAM_USED=$(echo $RAM_DATA  | awk '{print $2}')
  RAM_FREE=$(echo $RAM_DATA  | awk '{print $3}')
  SWAP_TOTAL=$(echo $RAM_DATA | awk '{print $4}')
  SWAP_USED=$(echo $RAM_DATA  | awk '{print $5}')

  # ── DISK ──
  DISK_JSON=$(get_disk_json)

  # ── NETWORK ──
  NET_NOW=$(get_net_stats)
  CUR_RX=$(echo $NET_NOW | awk '{print $1}')
  CUR_TX=$(echo $NET_NOW | awk '{print $2}')

  NET_UP=$(( (CUR_TX - PREV_TX) / INTERVAL ))
  NET_DOWN=$(( (CUR_RX - PREV_RX) / INTERVAL ))
  [ $NET_UP -lt 0 ] && NET_UP=0
  [ $NET_DOWN -lt 0 ] && NET_DOWN=0

  PREV_RX=$CUR_RX
  PREV_TX=$CUR_TX

  # ── UPTIME ──
  UPTIME_SEC=$(awk '{print int($1)}' /proc/uptime)

  # ── INSERT SNAPSHOT ──
  TIMESTAMP=$(date -u +%Y-%m-%dT%H:%M:%SZ)
  PAYLOAD="{
    \"server_id\": \"$SERVER_ID\",
    \"created_at\": \"$TIMESTAMP\",
    \"cpu_usage\": $CPU_USAGE,
    \"cpu_temp\": $CPU_TEMP,
    \"cpu_cores\": $CPU_CORES,
    \"ram_total\": $RAM_TOTAL,
    \"ram_used\": $RAM_USED,
    \"ram_free\": $RAM_FREE,
    \"swap_total\": $SWAP_TOTAL,
    \"swap_used\": $SWAP_USED,
    \"disk\": $DISK_JSON,
    \"net_up\": $NET_UP,
    \"net_down\": $NET_DOWN,
    \"net_total_up\": $CUR_TX,
    \"net_total_down\": $CUR_RX,
    \"uptime_sec\": $UPTIME_SEC
  }"

  supabase_insert "metrics_snapshots" "$PAYLOAD"

  # ── HISTORY (hanya setiap 12x = 1 menit) ──
  if [ $i -eq $LOOP_COUNT ]; then
    supabase_insert "metrics_history" "{
      \"server_id\": \"$SERVER_ID\",
      \"cpu_usage\": $CPU_USAGE,
      \"ram_used\": $RAM_USED,
      \"ram_total\": $RAM_TOTAL,
      \"net_up\": $NET_UP,
      \"net_down\": $NET_DOWN
    }"

    # Update bandwidth daily (upsert)
    TODAY=$(date -u +%Y-%m-%d)
    supabase_upsert "bandwidth_daily" "{
      \"server_id\": \"$SERVER_ID\",
      \"date\": \"$TODAY\",
      \"total_up\": $CUR_TX,
      \"total_down\": $CUR_RX
    }" "server_id,date"

    # Services (setiap menit)
    SERVICES_JSON=$(get_services_json)
    supabase_upsert "services_status" "{
      \"server_id\": \"$SERVER_ID\",
      \"services\": $SERVICES_JSON,
      \"updated_at\": \"$TIMESTAMP\"
    }" "server_id"
  fi

  sleep $INTERVAL
done
```

## 💻 IMPLEMENTASI VERCEL API ROUTE

### `/api/agent/[server_id]/[api_key]/route.ts`

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { createClient } from '@supabase/supabase-js';
import fs from 'fs';
import path from 'path';

const supabase = createClient(
  process.env.SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
);

export async function GET(
  req: NextRequest,
  { params }: { params: { server_id: string; api_key: string } }
) {
  const { server_id, api_key } = params;

  // Validasi server_id + api_key
  const { data: server, error } = await supabase
    .from('servers')
    .select('id')
    .eq('id', server_id)
    .eq('api_key', api_key)
    .single();

  if (error || !server) {
    return new NextResponse('Unauthorized', { status: 401 });
  }

  // Baca template script
  const templatePath = path.join(process.cwd(), 'public', 'agent-template.sh');
  let script = fs.readFileSync(templatePath, 'utf-8');

  // Inject credentials ke script
  script = script
    .replace(/{{SERVER_ID}}/g, server_id)
    .replace(/{{SUPABASE_URL}}/g, process.env.NEXT_PUBLIC_SUPABASE_URL!)
    .replace(/{{SUPABASE_ANON_KEY}}/g, process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!);

  return new NextResponse(script, {
    headers: {
      'Content-Type': 'text/plain',
      'Content-Disposition': 'attachment; filename="monitor-agent.sh"',
    },
  });
}
```

### `/api/servers/register/route.ts`

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { createClient } from '@supabase/supabase-js';
import { randomBytes } from 'crypto';

const supabase = createClient(
  process.env.SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
);

export async function POST(req: NextRequest) {
  const { name } = await req.json();

  const server_id = randomBytes(8).toString('hex');       // contoh: a1b2c3d4e5f67890
  const api_key   = randomBytes(16).toString('hex');      // 32 char hex

  const { error } = await supabase.from('servers').insert({
    id: server_id,
    name,
    api_key,
  });

  if (error) {
    return NextResponse.json({ error: error.message }, { status: 500 });
  }

  const baseUrl = process.env.NEXT_PUBLIC_APP_URL || 'https://your-app.vercel.app';

  const install_command =
    `script_name="monitor-agent.sh"; ` +
    `curl -fsSL "${baseUrl}/api/agent/${server_id}/${api_key}" -o "$PWD/$script_name" ` +
    `|| wget -O "$PWD/$script_name" "${baseUrl}/api/agent/${server_id}/${api_key}"; ` +
    `chmod +x "$PWD/$script_name"; ` +
    `(crontab -l 2>/dev/null | grep -v "$script_name"; ` +
    `echo "* * * * * /bin/bash $PWD/$script_name") | crontab -; ` +
    `echo "✅ Monitor berhasil diinstal!"`;

  return NextResponse.json({
    server_id,
    api_key,
    install_command,
  });
}
```

## 🔐 KEAMANAN INSTALL SYSTEM

| Aspek | Implementasi |
|-------|-------------|
| Script URL aman | URL berisi api_key unik → hanya pemilik yang bisa download |
| Credentials di script | `SUPABASE_ANON_KEY` (read-limited) + `SERVER_ID` (isolated) |
| RLS Policy | Agent hanya bisa INSERT ke server_id miliknya |
| Revoke akses | Hapus server dari dashboard → api_key invalid → script berhenti kerja |
| Re-install | Generate api_key baru → install ulang |
